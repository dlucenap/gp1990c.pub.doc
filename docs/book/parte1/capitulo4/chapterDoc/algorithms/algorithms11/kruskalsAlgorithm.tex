FUNCTION Krukal($G(V,E)$):
  $n$ $\leftarrow$ numNodes;
  $T = \phi$;
  DO
     $e$ $\leftarrow$ $\{u,v\}$
     nodeU $\leftarrow$ find($u$)
     nodeV $\leftarrow$ find($v$)
     IF (nodeU $\neq$ nodeV) THEN
        union(nodeU,nodeV)
        $T \leftarrow T \cup \{e\}$
  WHILE $T = n - 1$
  RETURN $T$
